```javascript JavaScript
import { Firecrawl } from 'firecrawl';

const app = new Firecrawl({apiKey: "fc-YOUR_API_KEY"});

// Crawl avec scraping mis en cache — 500 % plus rapide pour les pages vues récemment
const crawlResult = await app.crawlUrl('https://firecrawl.dev', {
  limit: 100,
  scrapeOptions: {
    formats: ['markdown'],
    maxAge: 3600000 // Utiliser les données en cache si elles datent de moins d'une heure
  }
});

crawlResult.data.forEach(page => {
  console.log(`URL: ${page.metadata.sourceURL}`);
  console.log(`Content: ${page.markdown.substring(0, 200)}...`);
});
```
