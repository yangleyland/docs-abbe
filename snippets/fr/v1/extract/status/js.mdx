```js Node
import { Firecrawl } from "firecrawl";

const app = new Firecrawl({
  apiKey: "fc-YOUR_API_KEY"
});

// Commencez par lancer une tâche d'extraction
const extractJob = await app.asyncExtract([
  'https://docs.firecrawl.dev/*', 
  'https://firecrawl.dev/'
], {
  prompt: "Extract the company mission and features from these pages."
});

// Récupérer l'état de la tâche d'extraction
const jobStatus = await app.getExtractStatus(extractJob.jobId);

console.log(jobStatus);
// Exemple de résultat :
// {
//   status: "terminée",
//   progress: 100,
//   results: [{
//     url: "https://docs.firecrawl.dev",
//     data: { ... }
//   }]
// }
```
