```js Node
import { Firecrawl } from "firecrawl";

// Initialiser le client
const firecrawl = new Firecrawl({ apiKey: "your_api_key" });

// Définir les paramètres de génération
const params = {
  maxUrls: 2, // Nombre maximal d'URL à analyser
  showFullText: true, // Inclure le texte intégral dans les résultats
};

// Générer LLMs.txt avec interrogations successives
const results = await firecrawl.generateLLMsText("https://example.com", params);

// Consulter les résultats de la génération
if (results.success) {
  console.log(`Status: ${results.status}`);
  console.log(`Generated Data:`, results.data);
} else {
  console.error(`Error: ${results.error || "Unknown error"}`);
}
```
