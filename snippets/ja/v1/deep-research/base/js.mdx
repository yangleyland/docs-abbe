```js Node
import { Firecrawl } from 'firecrawl';

// クライアントを初期化する
const firecrawl = new Firecrawl({ apiKey: 'your_api_key' });

// リサーチ用のパラメータを定義
const params = {
  maxDepth: 5,  // リサーチの反復回数
  timeLimit: 180,  // 制限時間（秒）
  maxUrls: 15  // 解析するURLの最大数
};

// リアルタイム更新でリサーチを開始
const onActivity = (activity) => {
  console.log(`[${activity.type}] ${activity.message}`);
};

// 深堀りリサーチを実行
const results = await firecrawl.deepResearch(
  'What are the latest developments in quantum computing?',
  params,
  onActivity
);

// リサーチ結果にアクセス
console.log(`Final Analysis: ${results.data.finalAnalysis}`);
console.log(`Sources: ${results.data.sources.length} references`); 
```
