```js Node
// npm install firecrawl

import { Firecrawl } from 'firecrawl';

const app = new Firecrawl({ apiKey: 'YOUR_API_KEY' });

// リトライロジック付きスクレイピング関数
async function scrapeWithRetry(url) {
  try {
    // まずデフォルトプロキシで試行
    const content = await app.scrapeUrl(url);
    
    // エラーステータスコードを受信したか確認
    const statusCode = content?.metadata?.statusCode;
    if ([401, 403, 500].includes(statusCode)) {
      console.log(`Got status code ${statusCode}, retrying with enhanced proxy`);
      // 拡張プロキシで再試行
      return await app.scrapeUrl(url, {
        proxy: 'enhanced'
      });
    }
    
    return content;
  } catch (error) {
    console.error(`Error: ${error.message}`);
    // 例外発生時に拡張プロキシで再試行
    try {
      return await app.scrapeUrl(url, {
        proxy: 'enhanced'
      });
    } catch (retryError) {
      console.error(`Enhanced proxy also failed: ${retryError.message}`);
      throw retryError;
    }
  }
}

// 使用例
const content = await scrapeWithRetry('https://example.com');
console.log(content.markdown);
```
