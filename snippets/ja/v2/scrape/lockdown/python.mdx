```python Python
from firecrawl import Firecrawl

firecrawl = Firecrawl(api_key="fc-YOUR_API_KEY")

# キャッシュされた結果のみを返す。外部へのリクエストは行われない。
# URLがキャッシュに存在しない場合は SCRAPE_LOCKDOWN_CACHE_MISS を返す。
scrape_result = firecrawl.scrape(
    'https://firecrawl.dev',
    formats=['markdown'],
    lockdown=True,
)

print(scrape_result.markdown)
```
